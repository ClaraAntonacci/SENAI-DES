CREATE DATABASE escola;

CREATE TABLE professores(
    id INTEGER AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    email VARCHAR(100),
    telefone VARCHAR(50)
);

CREATE TABLE disciplinas(
    id INTEGER AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    professorid INTEGER,
    FOREIGN KEY (professorid) REFERENCES professores(id)
);

CREATE TABLE turmas(
    id INTEGER AUTO_INCREMENT PRIMARY KEY,
    periodo VARCHAR(100),
    nome VARCHAR (100)
);

CREATE TABLE possui(
    turmasid INTEGER,
    disciplinasid INTEGER,
    FOREIGN KEY (turmasid) REFERENCES turmas(id),
    FOREIGN KEY (disciplinasid) REFERENCES disciplinas(id)
);
